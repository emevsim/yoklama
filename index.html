const express = require('express'); const app = express(); const http = require('http').createServer(app); // CORS ayarlarını en geniş şekilde açıyoruz const io = require('socket.io')(http, { cors: { origin: "*", // Tüm adreslerden gelen isteklere izin ver methods: ["GET", "POST"], credentials: true } }); // Gelen istekleri loglayalım (Hata ayıklamak için) io.on('connection', (socket) => { console.log('Yeni bağlantı:', socket.id); socket.on('join-room', (roomId) => { socket.join(roomId); console.log(`Odaya girildi: ${roomId}`); }); socket.on('yoklama-gonder', (data) => { console.log('Veri geldi:', data); socket.to(data.roomId).emit('yoklama-tetikle', data.ogrenciNo); }); }); const PORT = process.env.PORT || 3000; http.listen(PORT, () => { console.log(`Sunucu ${PORT} portunda aktif.`); });
